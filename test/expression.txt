1
==>
Nix(Int)

1+2
==>
Nix(Add(Int,Int))

# add with whitespace
1 + 2
==>
Nix(Add(Int,Int))

# line comment
# hello
1
==>
Nix(LineComment,Int)

# sub
1-2
==>
Nix(Sub(Int,Int))

# mul
1*2
==>
Nix(Mul(Int,Int))

# div
1/2
==>
Nix(Div(Int,Int))

# string
"a"
==>
Nix(String(StringContent))

# list empty
[]
==>
Nix(List)

# list int
[1]
==>
Nix(List(Int))

# list int string
[1 "a"]
==>
Nix(List(Int,String(StringContent)))

# list concat 2
[] ++ []
==>
Nix(ConcatList(List,List))

# list concat 3
[] ++ [] ++ []
==>
Nix(ConcatList(List,List,List))

# string concat 2
"a" + "b"
==>
Nix(Add(String(StringContent),String(StringContent)))

# string concat 3
"a" + "b" + "c"
==>
Nix(Add(String(StringContent),String(StringContent),String(StringContent)))
