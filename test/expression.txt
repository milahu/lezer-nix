# int
1
==>
Nix(Int)

# add
1+2
==>
Nix(Add(Int,Int))

# add with whitespace
1 + 2
==>
Nix(Add(Int,Int))

# line comment
# hello
1
==>
Nix(LineComment,Int)

# line comment precedence
#/*
1
#*/
==>
Nix(LineComment,Int,LineComment)

# block comment
/* hello */1
==>
Nix(BlockComment,Int)

# block comment precedence
/*
# hello
*/
==>
Nix(BlockComment)

# sub
1-2
==>
Nix(Sub(Int,Int))

# mul
1*2
==>
Nix(Mul(Int,Int))

# div
1/2
==>
Nix(Div(Int,Int))

# precedence mul add
1*2+3
==>
Nix(Add(Mul(Int,Int),Int))

# string raw
"a"
==>
Nix(String(StringRaw(StringRawContent)))

# string indent
''a''
==>
Nix(String(StringIndent(StringIndentContent)))

# list empty
[]
==>
Nix(List)

# list int
[1]
==>
Nix(List(Int))

# list int string
[1 "a"]
==>
Nix(List(Int,String(StringRaw(StringRawContent))))

# list concat 2
[] ++ []
==>
Nix(ConcatList(List,List))

# list concat 3
[] ++ [] ++ []
==>
Nix(ConcatList(List,List,List))

# string concat 2
"a" + "b"
==>
Nix(Add(String(StringRaw(StringRawContent)),String(StringRaw(StringRawContent))))

# string concat 3
"a" + "b" + "c"
==>
Nix(Add(String(StringRaw(StringRawContent)),String(StringRaw(StringRawContent)),String(StringRaw(StringRawContent))))

# attrset 0
{}
==>
Nix(AttrSet)

# attrset 2
{ "a" = 1; "b" = "2"; }
==>
Nix(AttrSet(Attr(String(StringRaw(StringRawContent)),Int),Attr(String(StringRaw(StringRawContent)),String(StringRaw(StringRawContent)))))

# attrpath string 2
"a"."b"
==>
Nix(AttrPath(String(StringRaw(StringRawContent)),String(StringRaw(StringRawContent))))

# attrpath string 3
"a"."b"."c"
==>
Nix(AttrPath(String(StringRaw(StringRawContent)),String(StringRaw(StringRawContent)),String(StringRaw(StringRawContent))))

# attrset with identifier
{ a = 1; }
==>
Nix(AttrSet(Attr(Identifier,Int)))
