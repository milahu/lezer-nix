# int
1
==>
Nix(Int)

# add
1+2
==>
Nix(Add(Int,Int))

# add with whitespace
1 + 2
==>
Nix(Add(Int,Int))

# line comment
# hello
1
==>
Nix(LineComment,Int)

# sub
1-2
==>
Nix(Sub(Int,Int))

# mul
1*2
==>
Nix(Mul(Int,Int))

# div
1/2
==>
Nix(Div(Int,Int))

# string
"a"
==>
Nix(String(StringContent))

# list empty
[]
==>
Nix(List)

# list int
[1]
==>
Nix(List(Int))

# list int string
[1 "a"]
==>
Nix(List(Int,String(StringContent)))

# list concat 2
[] ++ []
==>
Nix(ConcatList(List,List))

# list concat 3
[] ++ [] ++ []
==>
Nix(ConcatList(List,List,List))

# string concat 2
"a" + "b"
==>
Nix(Add(String(StringContent),String(StringContent)))

# string concat 3
"a" + "b" + "c"
==>
Nix(Add(String(StringContent),String(StringContent),String(StringContent)))

# attrset 0
{}
==>
Nix(AttrSet)

# attrset 2
{ "a" = 1; "b" = "2"; }
==>
Nix(AttrSet(Attr(String(StringContent),Int),Attr(String(StringContent),String(StringContent))))

# attrpath string 2
"a"."b"
==>
Nix(AttrPath(String(StringContent),String(StringContent)))

# attrpath string 3
"a"."b"."c"
==>
Nix(AttrPath(String(StringContent),String(StringContent),String(StringContent)))
