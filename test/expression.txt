''a''

''
  a${x}b
''
==>
Nix(StringBlock(StringBlockContent,StringBlockInterpolation(Identifier),StringBlockContent))

''a${x}b''
==>
Nix(StringBlock(StringBlockContent,StringBlockInterpolation(Identifier),StringBlockContent))

# string block with escaped interpolation single line
''a''${x}b''
==>
Nix(StringBlock(StringBlockContent))

''

''
==>
